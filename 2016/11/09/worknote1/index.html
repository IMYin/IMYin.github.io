代码¶

package cn.com.dtmobile.hadoop.biz.countmainorg.mr;

import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

import org.apache.hadoop.io.NullWritable;
import org.apache.hadoop.io.Text;
import org.apache.hadoop.mapreduce.Reducer;

import cn.com.dtmobile.hadoop.biz.countmainorg.util.DateUtils;
import cn.com.dtmobile.hadoop.biz.countmainorg.util.GetIndex;
import cn.com.dtmobile.hadoop.biz.countmainorg.util.UeMrKey;

public class UeMrSortReduce extends Reducer {
    @Override
    public void reduce(UeMrKey key, Iterable values, Context context)
            throws IOException, InterruptedException {
        int nei_num = 23;
        String tail = ",,,,,,,,,,,,,,,,,,,,,,,,,,,,";

        List rows = new ArrayList();
        for (Text value : values) {
            rows.add(new Text(value));
        }
        // backfill TA and AOA
        int id_list = 0;
        for (Text value_b : rows) {
            String[] word_b = value_b.toString().split(",");
            int index_of_TA = GetIndex.getIndex(value_b, 18);
            int index_of_AOA = GetIndex.getIndex(value_b, 20);

            String s_head = value_b.toString().substring(0, index_of_TA + 1);
            String s_tail = value_b.toString().substring(index_of_AOA);

            int t_b = DateUtils.convertSecond(word_b[13]) / 1000;
            String TA = "";
            String AOA = "";
            for (Text value_n : rows) {
                String[] word_n = value_n.toString().split(",");
                int t_n = DateUtils.convertSecond(word_n[13]) / 1000;
                if (word_b[11].equals(word_n[11]) && word_b[12].equals(word_n[12]) && t_n - t_b <= 30) {
                    if (TA.length() == 0) {TA = word_n[18];}
                    if (AOA.length() == 0) {AOA = word_n[19];}
                }else {continue;}
            }
            String str = s_head + TA + "," + AOA + s_tail;
            rows.set(id_list, new Text(str));
            id_list+=1;
        }
        // get the number of the column "NeighborCellNum"
        for (Text value : rows) {

            int index = GetIndex.getIndex(value, 24);
            // covert the column to row
            String[] words = value.toString().split(",");
            if (Integer.parseInt(words[nei_num]) == 0) {
                StringBuffer sb = new StringBuffer();
                sb.append(value.toString().substring(0));
                context.write(NullWritable.get(), new Text(sb.toString()));
            } else {
                for (int i = 0; i < Integer.parseInt(words[nei_num]); i++) {
                    int k = nei_num + 4 * i;
                    StringBuffer sb = new StringBuffer();
                    sb.append(value.toString().substring(0, index)); 
                    sb.append(",");
                    sb.append(words[k + 1]);
                    sb.append(",");
                    sb.append(words[k + 2]);
                    sb.append(",");
                    sb.append(words[k + 3]);
                    sb.append(",");
                    sb.append(words[k + 4]);
                    sb.append(tail);
                    context.write(NullWritable.get(), new Text(sb.toString())); //让key值为NULL
                }
            }
        }
    }
}

返回字符串中第N个","的索引值¶

按代码的运行顺序走吧，从上往下罗列。

在开始我用到了一个方法：GetIndex.getIndex().

首先介绍一下背景：我们的输入数据是表的形式，字段之间的分隔符是","，后面为了替换某一字段的内容，在这里需要把前面的字符串截出来，所以我需要获取某一字段的索引值。我的方法是这么写的：

package cn.com.dtmobile.hadoop.biz.countmainorg.util;

import java.util.regex.Matcher;
import java.util.regex.Pattern;

import org.apache.hadoop.io.Text;

public class GetIndex {
    public static int getIndex(Text value, int num) {
        int index = 0;
        int n = 0;
        Pattern p = Pattern.compile(",");
        Matcher m = p.matcher(value.toString());
        while (m.find()) {
            n++;
            if (num == n) {
                index = m.start();
            }
        }
        return index;
    }
}

我用了正则表达式匹配出","，然后找出第Ｎ个","，返回其索引值。

对表进行列转行处理¶

下面就进入了我的第二个业务逻辑，对表进行列转行。

背景是这样的：字段A的类型为int，表示后面有几组数据，比如是2的话，就代表有2组数据，每组数据中有4个字段。如下：

def backfill(line):
    lst = []
    if (int(line[0])==0):
        lst.append(line)
        return lst
    else:
        for i in range(int(line[0])):
            lst_a = line.split(",")
            for j in range(1,5):
                lst_a[j] = lst_a[j+i*4]
            s = ",".join(lst_a)
            lst.append(s)
        return lst
for i in [item for d in [a,b,c] for item in backfill(d)]:

首先我初始化一个StringBuffer()，然后赋值于前面相同的值。在后面，我用append()方法讲列转行的内容补充进去，剩余没有补充的元素，全部赋值为空，并且用","补齐，以便入库。

这么做，有几个好处：

  • 不用(if else)判断后面有几个组，只要循环就行，代码显得优美，如果后面有100个组，我也用担心；

  • 如果用clone()方法也可以，但是如果用StringBuffer()则更加节省资源，效率高；

  • 数据后面全为空，也为之后入库节约资源。
